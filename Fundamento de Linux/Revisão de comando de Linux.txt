* apt - GErenciador de Pacotes 
* apt install - instalarPacotes 
* apt upgrade - AtualizaPacotes
* apt Update && apt apgrade - atualiza o sistema
* dpkg - gerenciador de pacotes deb
* dpkg -i - instalar um pacote deb 
* dpkg -r - remove um pacote deb
* dpkg -L - informações do pacotes deb
* rpm - Gerenciador de Pacote deb
* rpm -ivh --nodes  - instala um pacote rpm
* rpm -U  - atualiza um pacote rpm
* rpm -e  - remove um pacote rpm
* yum Gerenciador de pacote rpm
* yum install - instala um pacote rpm
* yum update - atualiza um pacote rpm
* yum remove - remove um pacote rpm
